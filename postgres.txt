                   user_id int PRIMARY KEY,
                   username VARCHAR NOT NULL,
                   password INTEGER NOT NULL,
                   phone VARCHAR (64)  NOT NULL,
                   email VARCHAR (64)  NOT NULL,
                   country country_type,
                   city city_type
                                        );
